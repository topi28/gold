// Interfacing of LED with 8051

org 0000H
MOV A, #38H;
ACALL COM;
MOV A, #83H;
ACALL COM;
MOV A, #01H;
MOV DPTR, #STR;
CLR A;
MOV CA, @A+DPTR;
JZ FINISH;
ACALL LD-DATA;
INC DPTR;
SJMP REV;
SJMP FINISH;
ACALL DELAY;
MOV P2, A;
CLR P3.1;
SET B P3.2;
CLR P3.2;
RET
MOV R2, #0FFH;
MOV R3, #0FFH;
DJNZ R3, HERE;
DJNZ R2, AGAIN;
RET;
DB 'VIDYALANKAR',0;
END;
